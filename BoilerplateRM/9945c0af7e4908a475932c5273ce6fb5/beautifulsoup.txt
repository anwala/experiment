    ss.rel = "stylesheet";
    ss.href = href.trim();

    ref.parentNode.appendChild( ss, ref );
    function toggleMedia(){
      var defined;
      for( var i = 0; i < sheets.length; i++ ){
        if( sheets[ i ].href && sheets[ i ].href.indexOf( href.trim().substr(-8)) > -1 ){
          defined = true;
        }
      }
      if( defined ){
        ss.media = media || "all";
      }
      else {
        setTimeout( toggleMedia );
      }
    }

    toggleMedia();
    return ss;
  }
